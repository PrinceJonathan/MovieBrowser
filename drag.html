$(document).ready(function() { //Click Wish Button to Save $('#wish-btn').click(function(){ saveData(); }); // 定義 savedata 的動作， function saveData(){ //如果 giftname 空白不能送出 if ( $('#giftname').val() == '') { alert('Please Write A Gift'); } else { //取得新願望 並塞到前台。 var newWish = $('#giftname').val(); newWish = escapeHtml(newWish); // 2014.09.19 更新，用於替換一些符號; //把data 存回去 $.ajax({ url: "addData.php", type: "POST", data: { wish : newWish }, success: function(data){ if ( data == '1'){ alert('哎呀，好像有什麼東西出錯啦，請稍後再試。'); } else { // do something if success } $('#giftname').val(''); //清空input } }); } }; }); //2014.09.19 更新, 用於取代一些符號; function escapeHtml(text) { return text .replace(/&/g, “&") .replace(//g, “>") .replace(/"/g, “"") .replace(/’/g, “'"); }
